

# Instructions for setting up Deep Learning Amazon AMI (ubuntu) can be found on this amazon blog post https://aws.amazon.com/blogs/machine-learning/get-started-with-deep-learning-using-the-aws-deep-learning-ami/


# Change permissions for private key to read only for ssh to work

chmod 400 DeepLearning.pem

# Then automatically initialise this private key for ssh

ssh-add DeepLearning.pem

# ssh to aws instance where DNS: ec2-54-244-162-101.us-west-2.compute.amazonaws.com

ssh -i "Deep_Learning.pem" ubuntu@ec2-54-244-162-101.us-west-2.compute.amazonaws.com


source activate tensorflow_p36  # or whatever environment

# To copy files from local hard disk to AWS instance - can use SCP protocol from terminal
# scp -i “.pem_filename” “path/to/file” user@hostname:path/where/file/to/be/copied/on_server

# can drop the -i if we do not want to include the private key.pem

scp /Users/ryannazareth/Documents/PyCon_2018/LIME.ipynb ubuntu@ec2-54-244-162-101.us-west-2.compute.amazonaws.com:~
